    \pageref{#2}--\pageref{#1}%
  \else
    \pageref{#1}%
    \ifnum\value{mystart}<\value{mystop}%
      \addtocounter{mystop}{-1}%
      \ifnum\value{mystart}<\value{mystop}%
        \,ff.
        %--\pageref{#2}%%
      \else
        \,f.
         %%--\pageref{#2}%
              \fi
    \fi
  \fi
}
